// FunciÓN: ENVÍO SPI (Parte 2)
// --------------------------------------------------------------------------
void enviar_comando_spi(char *comando){
    uint8_t len = strlen(comando);

    transmit_uart("\r\n[SPI] Enviando comando: ");
    transmit_uart(comando);
    transmit_uart("\r\n");

    // Activar Chip Select (CS)
    HAL_GPIO_WritePin(SD_SS_GPIO_Port, SD_SS_Pin, GPIO_PIN_RESET);

    // Transmitir el comando por SPI1
    HAL_StatusTypeDef status = HAL_SPI_Transmit(&hspi1, (uint8_t*)comando, len, 500);
    HAL_Delay(5);
    //  Desactivar Chip Select (CS)
    HAL_GPIO_WritePin(SD_SS_GPIO_Port, SD_SS_Pin, GPIO_PIN_SET);

    if (status == HAL_OK) {
        transmit_uart(" Comando enviado correctamente del SPI a ESP32 ejecutando...\r\n");
    } else {
        transmit_uart(" ERROR al enviar comando SPI.\r\n");
    }
}

// FunciÓN: ACCIÓN SPI

void accionSPI(void){
	char comando_buffer[32]; // Buffer para almacenar el comando del usuario (ej: "1,500")
	    uint8_t len = 0;
		transmit_uart("\r\n=====================================\r\n");
	    transmit_uart("\r\n--- SUB-MENU CONTROL SPI ---\r\n");
		transmit_uart("\r\n=====================================\r\n");
	    transmit_uart("Formato: LED,Tiempo_ms (Ej: 1,500)\r\n");
	    transmit_uart("LEDs disponibles: 1, 2, 3\r\n");
		transmit_uart("\r\n");
		transmit_uart("\r\n=====================================\r\n");
		transmit_uart("\r\n");
	    transmit_uart("Ingrese comando: ");

	    // 1. Leer el comando completo (cadena de caracteres)
	    // Bucle para capturar la entrada hasta que el usuario presione ENTER

	    while(len < sizeof(comando_buffer) - 1)
	    {
	        uint8_t temp_rx;
	        // Espera a recibir un caracter
	        HAL_UART_Receive(&huart2, &temp_rx, 1, HAL_MAX_DELAY);

	        if (temp_rx == '\r' || temp_rx == '\n') {

	            // Consumir el siguiente caracter si es el par CR+LF
	            uint8_t next_char;
	            if (HAL_UART_Receive(&huart2, &next_char, 1, 100) == HAL_OK) {
	                // Si es el par, descartamos el segundo caracter.
	                if ((temp_rx == '\r' && next_char == '\n') || (temp_rx == '\n' && next_char == '\r')) {
	                     // Ya se consumió y se ignora.
	                } else {

	                }
	            }
	            break; // Salir del bucle de lectura
	        }


	        HAL_UART_Transmit(&huart2, &temp_rx, 1, 100);

	        // Guardar en el buffer
	        comando_buffer[len++] = (char)temp_rx;
	    }

	    // Finalizar la cadena de caracteres
	    comando_buffer[len] = '\0';
        transmit_uart("\r\n");
	    // Validar y Enviar
	    if (len > 0) {
	        // Enviar la cadena de comando al ESP32 por SPI
	        enviar_comando_spi(comando_buffer);
	    } else {
	        transmit_uart("\r\n[SPI] Comando vacio Cancelando operacion.\r\n");
	    }
}

// I2C
// --------------------------------------------------------------------------
void accionI2C(void){
    // dirección de 8 bits 
    uint8_t SlaveAddr_Read  = (deviceAddress << 1) | 0x01; // Dirección para Lectura (0x61)
    uint8_t RxData[2]; // Buffer de 2 bytes, porque el ESP32 envía un uint16_t (2 bytes)
    HAL_StatusTypeDef status;

    // 1. Solicitud de 2 bytes (uint16_t) de medición
    transmit_uart("\r\n--- INICIO MEDICION I2C ---\r\n");
    // Usamos el buffer global 'buffer' para evitar el error de compilación
    sprintf(buffer, "[I2C] Solicitando 2 bytes (ADC uint16) al ESP32 (0x%X)...\r\n", deviceAddress);
    transmit_uart(buffer);

    // Solicitamos 2 bytes. El ESP32 responderá con adcToSend.
    status = HAL_I2C_Master_Receive(&hi2c1, (uint16_t)SlaveAddr_Read, RxData, 2, 500);

    if (status == HAL_OK) {
        // Unimos los dos bytes recibidos para formar el uint16_t (Little Endian)
        uint16_t adc_value = (uint16_t)RxData[0] | ((uint16_t)RxData[1] << 8);

        // Cálculo del Voltaje (3.3V / 4095)
        float voltage = ((float)adc_value / 255.0f) * 3.3f;

        transmit_uart("\r\n[I2C] Recepcion EXITOSA:\r\n");

        // Usamos el buffer global 'buffer'
        sprintf(buffer, "  -> ADC (bits): %hu\r\n", adc_value);
        transmit_uart(buffer);

        // Usamos el buffer global 'buffer'
        //sprintf(buffer, "  -> Voltaje (V): %.2f\r\n", voltage);
        //transmit_uart(buffer);
        // WORKAROUND para evitar error %f de sprintf en STM32/Newlib-nano
                int integerPart = (int)voltage;
                // Obtenemos los decimales (los 3 primeros)
                int fractionalPart = (int)((voltage - integerPart) * 1000.0f);

                // Cargar el valor de voltaje al buffer para impresion (sin usar %f)
                // Usamos %d.%03d para garantizar 3 decimales
                sprintf((char*)buffer, " -> Voltaje (V): %d.%01d\r\n", integerPart, fractionalPart);
                transmit_uart(buffer);

    } else {
        // Fallo en la comunicación I2C (HAL_TIMEOUT, HAL_ERROR, etc.)
        transmit_uart("\r\n[I2C] ERROR: Fallo al recibir la medicion I2C. ");
        // Imprimir el estado del error para diagnóstico
        // Usamos el buffer global 'buffer'
        sprintf(buffer, "Status: %d\r\n", status);
        transmit_uart(buffer);
    }

    HAL_Delay(500);
}
